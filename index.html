Control de Flujo y Manejo de Errores

if...else


if (condition) {
    // código a ejecutar si la condición es verdadera
} else {
    // código a ejecutar si la condición es falsa
}

switch


switch (expression) {
    case value1:
        // código a ejecutar si la expresión es igual a value1
        break;
    case value2:
        // código a ejecutar si la expresión es igual a value2
        break;
    default:
        // código a ejecutar si no coincide con ningún valor
}

try/catch/throw


try {
    // código que puede lanzar una excepción
    throw new Error('Algo salió mal');
} catch (error) {
    console.error(error.message);
}

Objetos Error


const error = new Error('Mensaje de error');
console.log(error.name);    // 'Error'
console.log(error.message); // 'Mensaje de error'

Bucles e Iteración

for


for (let i = 0; i < 10; i++) {
    console.log(i);
}

while


let i = 0;
while (i < 10) {
    console.log(i);
    i++;
}

do...while


let i = 0;
do {
    console.log(i);
    i++;
} while (i < 10);

Funciones

Definir funciones


function saludar(nombre) {
    return 'Hola ' + nombre;
}

Llamar funciones


const mensaje = saludar('Mundo');
console.log(mensaje); // 'Hola Mundo'

Expresiones y Operadores

Asignación y Comparación


let x = 10; // Asignación
let y = 20;
if (x === y) {
    console.log('x es igual a y');
} else {
    console.log('x no es igual a y');
}

Operadores aritméticos


let suma = 10 + 5;
let resta = 10 - 5;
let multiplicacion = 10 * 5;
let division = 10 / 5;
console.log(suma, resta, multiplicacion, division);

Operadores lógicos


let a = true;
let b = false;
console.log(a && b); // false
console.log(a || b); // true
console.log(!a);     // false

Números y Fechas

Objeto Number


let numero = 123.456;
console.log(numero.toFixed(2)); // '123.46'

Objeto Math


let raizCuadrada = Math.sqrt(16);
let maximo = Math.max(10, 20, 30);
console.log(raizCuadrada, maximo);

Objeto Date


let fecha = new Date();
console.log(fecha.toDateString()); // 'Sat Jul 07 2024'

Formateo de Texto

Cadenas literales


let nombre = 'Juan';
let saludo = `Hola, ${nombre}`;
console.log(saludo); // 'Hola, Juan'

Objeto String


let texto = 'JavaScript';
console.log(texto.length);      // 10
console.log(texto.toUpperCase()); // 'JAVASCRIPT'

Colecciones Indexadas

Arreglos


let numeros = [1, 2, 3, 4, 5];
numeros.forEach(num => console.log(num));

Arreglos tipados


let buffer = new ArrayBuffer(16);
let vista = new Int32Array(buffer);
vista[0] = 42;
console.log(vista[0]); // 42

DOM

Encontrar elementos HTML por ID


let elemento = document.getElementById('miElemento');
console.log(elemento);

Encontrar elementos HTML por nombre de etiqueta


let elementos = document.getElementsByTagName('p');
console.log(elementos);

Encontrar elementos HTML por nombre de clase


let elementos = document.getElementsByClassName('miClase');
console.log(elementos);

Encontrar elementos HTML mediante selectores CSS


let elemento = document.querySelector('.miClase');
console.log(elemento);
let elementos = document.querySelectorAll('.miClase');
console.log(elementos);

Encontrar elementos HTML por colecciones de objetos HTML


let coleccion = document.forms['miFormulario'];
console.log(coleccion);
